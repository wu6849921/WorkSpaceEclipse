function gettextinterval(b) {
	var d = Math.floor(b / 24);
	b -= d * 24;
	var a = Math.floor(b);
	b -= a;
	var c = (b * 60);
	c = c.toFixed(0);
	if (d == 0) {
		if (a == 0) {
			return c + " min "
		} else {
			return a + " hrs " + c + " min "
		}
	} else {
		if (a == 0) {
			if (c == 0) {
				return d + " days "
			} else {
				return d + " days " + a + " hrs " + c + " min "
			}
		} else {
			return d + " days " + a + " hrs " + c + " min "
		}
	}
}
